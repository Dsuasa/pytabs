265    def get_database_units(self) -> etabs.eUnits:
266        """Returns a value from the eUnits enumeration indicating the database units for the model.
267        All data is internally stored in the model in these units and converted to the present units as needed.
268
269        :raises EtabsError: Database units could not be returned
270        :return: Units enumeration
271        :rtype: eUnits
272        """
273        ret = self.sap_model.GetDatabaseUnits()
274        if ret == 0:
275            raise EtabsError(0, "Database units could not be returned.")
276        else:
277            return ret

280    def get_database_units_components(self) -> UnitsComponents:
281        """Retrieves the database units for the model.
282        All data is internally stored in the model in these units and converted to the present units as needed.
283
284        :return: units for force, length and temperature units
285        :rtype: UnitsComponents
286        """
287        force_units = etabs.eForce.NotApplicable
288        length_units = etabs.eLength.NotApplicable
289        temperature_units = etabs.eTemperature.NotApplicable
290        [ret, force_units, length_units, temperature_units] =  self.sap_model.GetDatabaseUnits_2(force_units, length_units, temperature_units)
291        handle(ret)
292        return {'force_units': force_units,
293                'length_units': length_units,
294                'temperature_units': temperature_units}
